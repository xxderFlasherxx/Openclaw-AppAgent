using UnityEngine;

/// <summary>
/// Fahrzeug-Controller mit WheelColliders.
/// Realistisches Fahrverhalten mit Beschleunigung, Bremsen und Lenkung.
/// </summary>
[RequireComponent(typeof(Rigidbody))]
public class VehicleController : MonoBehaviour
{
    [Header("Wheel Colliders")]
    [SerializeField] private WheelCollider frontLeftWheel;
    [SerializeField] private WheelCollider frontRightWheel;
    [SerializeField] private WheelCollider rearLeftWheel;
    [SerializeField] private WheelCollider rearRightWheel;

    [Header("Wheel Meshes (Visual)")]
    [SerializeField] private Transform frontLeftMesh;
    [SerializeField] private Transform frontRightMesh;
    [SerializeField] private Transform rearLeftMesh;
    [SerializeField] private Transform rearRightMesh;

    [Header("Vehicle Settings")]
    [SerializeField] private float maxMotorTorque = 1500f;
    [SerializeField] private float maxBrakeTorque = 3000f;
    [SerializeField] private float maxSteerAngle = 35f;
    [SerializeField] private float maxSpeed = 150f; // km/h

    [Header("Engine")]
    [SerializeField] private AnimationCurve torqueCurve;
    [SerializeField] private float downforce = 100f;

    private Rigidbody _rb;
    private float _steerInput;
    private float _throttleInput;
    private float _brakeInput;
    private bool _handbrake;

    /// <summary>Aktuelle Geschwindigkeit in km/h</summary>
    public float CurrentSpeedKmh => _rb.linearVelocity.magnitude * 3.6f;

    /// <summary>Aktuelle Geschwindigkeit normalisiert (0-1)</summary>
    public float SpeedNormalized => Mathf.Clamp01(CurrentSpeedKmh / maxSpeed);

    /// <summary>Motor-RPM (vereinfacht)</summary>
    public float EngineRPM { get; private set; }

    /// <summary>Ob das Fahrzeug am Boden ist</summary>
    public bool IsGrounded { get; private set; }

    private void Awake()
    {
        _rb = GetComponent<Rigidbody>();
        _rb.centerOfMass = new Vector3(0f, -0.5f, 0.2f);

        if (torqueCurve == null || torqueCurve.length == 0)
        {
            torqueCurve = new AnimationCurve(
                new Keyframe(0f, 1f),
                new Keyframe(0.5f, 1f),
                new Keyframe(1f, 0.3f)
            );
        }
    }

    private void Update()
    {
        // Input sammeln
        _steerInput = Input.GetAxis("Horizontal");
        _throttleInput = Mathf.Clamp01(Input.GetAxis("Vertical"));
        _brakeInput = Mathf.Clamp01(-Input.GetAxis("Vertical"));
        _handbrake = Input.GetKey(KeyCode.Space);

        // Wheel-Meshes aktualisieren
        UpdateWheelMesh(frontLeftWheel, frontLeftMesh);
        UpdateWheelMesh(frontRightWheel, frontRightMesh);
        UpdateWheelMesh(rearLeftWheel, rearLeftMesh);
        UpdateWheelMesh(rearRightWheel, rearRightMesh);
    }

    private void FixedUpdate()
    {
        ApplySteering();
        ApplyMotor();
        ApplyBraking();
        ApplyDownforce();
        CheckGrounded();
        CalculateEngineRPM();
    }

    private void ApplySteering()
    {
        // Geschwindigkeitsabhängige Lenkung (weniger bei hoher Geschwindigkeit)
        float speedFactor = Mathf.Lerp(1f, 0.4f, SpeedNormalized);
        float steerAngle = _steerInput * maxSteerAngle * speedFactor;

        frontLeftWheel.steerAngle = steerAngle;
        frontRightWheel.steerAngle = steerAngle;
    }

    private void ApplyMotor()
    {
        if (CurrentSpeedKmh >= maxSpeed)
        {
            rearLeftWheel.motorTorque = 0f;
            rearRightWheel.motorTorque = 0f;
            return;
        }

        float torqueMultiplier = torqueCurve.Evaluate(SpeedNormalized);
        float motorTorque = _throttleInput * maxMotorTorque * torqueMultiplier;

        rearLeftWheel.motorTorque = motorTorque;
        rearRightWheel.motorTorque = motorTorque;
    }

    private void ApplyBraking()
    {
        float brakeForce = _brakeInput * maxBrakeTorque;

        if (_handbrake)
        {
            // Handbremse nur auf Hinterräder
            rearLeftWheel.brakeTorque = maxBrakeTorque;
            rearRightWheel.brakeTorque = maxBrakeTorque;
            frontLeftWheel.brakeTorque = 0f;
            frontRightWheel.brakeTorque = 0f;
        }
        else
        {
            frontLeftWheel.brakeTorque = brakeForce;
            frontRightWheel.brakeTorque = brakeForce;
            rearLeftWheel.brakeTorque = brakeForce * 0.6f;
            rearRightWheel.brakeTorque = brakeForce * 0.6f;
        }
    }

    private void ApplyDownforce()
    {
        _rb.AddForce(-transform.up * downforce * SpeedNormalized);
    }

    private void CheckGrounded()
    {
        WheelHit hit;
        IsGrounded = frontLeftWheel.GetGroundHit(out hit)
                  || frontRightWheel.GetGroundHit(out hit)
                  || rearLeftWheel.GetGroundHit(out hit)
                  || rearRightWheel.GetGroundHit(out hit);
    }

    private void CalculateEngineRPM()
    {
        float wheelRPM = (rearLeftWheel.rpm + rearRightWheel.rpm) * 0.5f;
        EngineRPM = Mathf.Abs(wheelRPM) * 3.5f + 800f; // Vereinfacht
        EngineRPM = Mathf.Clamp(EngineRPM, 800f, 7000f);
    }

    private void UpdateWheelMesh(WheelCollider collider, Transform meshTransform)
    {
        if (collider == null || meshTransform == null) return;

        Vector3 pos;
        Quaternion rot;
        collider.GetWorldPose(out pos, out rot);
        meshTransform.position = pos;
        meshTransform.rotation = rot;
    }
}

using UnityEngine;
using UnityEngine.Events;
using System.Collections.Generic;
using System.Linq;

/// <summary>
/// Crafting-System für Sandbox-Spiele.
/// Verwendet ScriptableObject-basierte Rezepte.
/// </summary>
public class CraftingSystem : MonoBehaviour
{
    [Header("Recipes")]
    [SerializeField] private CraftingRecipe[] availableRecipes;

    [Header("References")]
    [SerializeField] private InventorySystem inventory;

    [Header("Events")]
    public UnityEvent<CraftingRecipe> OnItemCrafted;
    public UnityEvent<CraftingRecipe, string> OnCraftFailed; // recipe, reason

    /// <summary>Alle verfügbaren Rezepte</summary>
    public IReadOnlyList<CraftingRecipe> Recipes => availableRecipes;

    /// <summary>Prüft ob ein Rezept herstellbar ist.</summary>
    public bool CanCraft(CraftingRecipe recipe)
    {
        if (recipe == null || inventory == null) return false;

        foreach (var ingredient in recipe.ingredients)
        {
            if (!inventory.HasItem(ingredient.item, ingredient.count))
                return false;
        }
        return true;
    }

    /// <summary>Stellt ein Item her. Gibt true zurück bei Erfolg.</summary>
    public bool Craft(CraftingRecipe recipe)
    {
        if (recipe == null)
        {
            OnCraftFailed?.Invoke(recipe, "Rezept ist null.");
            return false;
        }

        if (!CanCraft(recipe))
        {
            OnCraftFailed?.Invoke(recipe, "Nicht genug Materialien.");
            return false;
        }

        // Prüfe ob Platz im Inventar
        if (inventory.FreeSlots < 1)
        {
            OnCraftFailed?.Invoke(recipe, "Inventar voll.");
            return false;
        }

        // Materialien entfernen
        foreach (var ingredient in recipe.ingredients)
        {
            inventory.RemoveItem(ingredient.item, ingredient.count);
        }

        // Ergebnis hinzufügen
        int leftover = inventory.AddItem(recipe.result, recipe.resultCount);
        if (leftover > 0)
        {
            Debug.LogWarning($"[Crafting] {leftover}x {recipe.result.itemName} konnte nicht hinzugefügt werden.");
        }

        OnItemCrafted?.Invoke(recipe);
        Debug.Log($"[Crafting] Hergestellt: {recipe.resultCount}x {recipe.result.itemName}");
        return true;
    }

    /// <summary>Gibt alle herstellbaren Rezepte zurück.</summary>
    public List<CraftingRecipe> GetCraftableRecipes()
    {
        return availableRecipes.Where(r => CanCraft(r)).ToList();
    }

    /// <summary>Gibt alle Rezepte für ein bestimmtes Ergebnis-Item zurück.</summary>
    public List<CraftingRecipe> GetRecipesForItem(ItemData resultItem)
    {
        return availableRecipes.Where(r => r.result == resultItem).ToList();
    }
}

/// <summary>
/// ScriptableObject für Crafting-Rezepte.
/// Erstelle im Unity Editor: Assets → Create → Game → CraftingRecipe
/// </summary>
[CreateAssetMenu(fileName = "NewRecipe", menuName = "Game/Crafting Recipe")]
public class CraftingRecipe : ScriptableObject
{
    [Header("Recipe Info")]
    public string recipeName;
    [TextArea(1, 3)]
    public string description;
    public Sprite icon;

    [Header("Ingredients")]
    public CraftingIngredient[] ingredients;

    [Header("Result")]
    public ItemData result;
    public int resultCount = 1;

    [Header("Requirements")]
    public CraftingStation requiredStation = CraftingStation.None;
    public int requiredLevel = 0;
}

/// <summary>Eine Zutat für ein Crafting-Rezept.</summary>
[System.Serializable]
public class CraftingIngredient
{
    public ItemData item;
    public int count = 1;
}

/// <summary>Crafting-Stationen (erweiterbarer Enum).</summary>
public enum CraftingStation
{
    None,        // Kann überall hergestellt werden
    Workbench,
    Furnace,
    Anvil,
    Alchemy
}

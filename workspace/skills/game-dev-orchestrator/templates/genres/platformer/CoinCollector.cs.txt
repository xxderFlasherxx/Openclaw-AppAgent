using UnityEngine;
using UnityEngine.Events;

/// <summary>
/// Collectible-System für 2D Platformer.
/// Münzen/Items die vom Spieler eingesammelt werden können.
/// </summary>
public class CoinCollector : MonoBehaviour
{
    [Header("Score")]
    [SerializeField] private int score = 0;
    [SerializeField] private int totalCoins = 0;

    [Header("Events")]
    public UnityEvent<int> OnScoreChanged;
    public UnityEvent<int, int> OnCoinCollected; // current, total

    /// <summary>Aktueller Score</summary>
    public int Score => score;

    /// <summary>Gesammelte Münzen</summary>
    public int TotalCoins => totalCoins;

    /// <summary>Fügt Punkte hinzu (aufgerufen von Collectible).</summary>
    public void AddScore(int points)
    {
        score += points;
        OnScoreChanged?.Invoke(score);
    }

    /// <summary>Registriert eine eingesammelte Münze.</summary>
    public void CollectCoin(int value = 1)
    {
        totalCoins++;
        AddScore(value);
        OnCoinCollected?.Invoke(totalCoins, value);
    }

    /// <summary>Setzt den Score zurück.</summary>
    public void ResetScore()
    {
        score = 0;
        totalCoins = 0;
        OnScoreChanged?.Invoke(score);
    }
}

/// <summary>
/// Einzelnes Collectible-Objekt. Auf die Münze/das Item platzieren.
/// </summary>
[RequireComponent(typeof(Collider2D))]
public class Collectible : MonoBehaviour
{
    [Header("Settings")]
    [SerializeField] private int scoreValue = 10;
    [SerializeField] private bool isCoin = true;
    [SerializeField] private bool destroyOnCollect = true;

    [Header("Animation")]
    [SerializeField] private float bobSpeed = 2f;
    [SerializeField] private float bobHeight = 0.2f;
    [SerializeField] private float rotateSpeed = 90f;

    [Header("Effects")]
    [SerializeField] private AudioClip collectSound;
    [SerializeField] private GameObject collectEffect;

    private Vector3 _startPos;

    private void Start()
    {
        _startPos = transform.position;
        GetComponent<Collider2D>().isTrigger = true;
    }

    private void Update()
    {
        // Schwebende Animation
        float newY = _startPos.y + Mathf.Sin(Time.time * bobSpeed) * bobHeight;
        transform.position = new Vector3(transform.position.x, newY, transform.position.z);

        // Rotation
        if (rotateSpeed > 0f)
        {
            transform.Rotate(Vector3.forward, rotateSpeed * Time.deltaTime);
        }
    }

    private void OnTriggerEnter2D(Collider2D other)
    {
        CoinCollector collector = other.GetComponent<CoinCollector>();
        if (collector == null) return;

        // Score hinzufügen
        if (isCoin)
            collector.CollectCoin(scoreValue);
        else
            collector.AddScore(scoreValue);

        // Sound
        if (collectSound != null)
        {
            AudioSource.PlayClipAtPoint(collectSound, transform.position);
        }

        // Effekt
        if (collectEffect != null)
        {
            Instantiate(collectEffect, transform.position, Quaternion.identity);
        }

        if (destroyOnCollect)
        {
            Destroy(gameObject);
        }
    }
}
